def basecount(seq): 
    """Count the frequencies of each bases in sequence including every letter""" 
    # {A: 3 30, C: 50, G: 4} 
    freq = {}
    for base in seq: 
        if base in freq: 
            freq[base] += 1 
        else: 
            freq[base] = 1 
    return freq

def basecount(seq, useall=False, calfreqs_pc=False): 
    """Count the frequencies of each bases in sequence including every letter"""
    
    length = len(seq)
    
    if calfreqs_pc: 
        freq_pc = {}
    else: 
        base_counts = {} 
    
    if useall: 
        seqset = set(seq) 
    else: 
        seqset = ("A", "T", "C", "G") 
        
    
    for letter in seqset: 
        num = seq.count(letter)
        if calfreqs_pc:
            pc = num / length
            freq_pc[letter] = pc
        else: 
            base_counts[letter] = num 
    
    
    if calfreqs_pc: 
        return freq_pc
    else: 
        return base_counts
